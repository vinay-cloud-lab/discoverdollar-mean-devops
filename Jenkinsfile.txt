pipeline {
    agent any

    stages {

        stage('Clone Code') {
            steps {
                git branch: 'main',
                    url: 'https://github.com/vinay-cloud-lab/discoverdollar-mean-devops.git'
            }
        }

        stage('Build Docker Images') {
            steps {
                sh 'docker build -t vinaychannalli/mean-backend:latest ./backend'
                sh 'docker build -t vinaychannalli/mean-frontend:latest ./frontend'
            }
        }

        stage('Docker Login') {
            steps {
                withCredentials([usernamePassword(credentialsId: 'dockerhub-creds', usernameVariable: 'USER', passwordVariable: 'PASS')]) {
                    sh 'echo $PASS | docker login -u $USER --password-stdin'
                }
            }
        }

        stage('Push Images') {
            steps {
                sh 'docker push vinaychannalli/mean-backend:latest'
                sh 'docker push vinaychannalli/mean-frontend:latest'
            }
        }

        stage('Deploy') {
            steps {
                sh 'docker-compose down'
                sh 'docker-compose up -d'
            }
        }
    }
}
